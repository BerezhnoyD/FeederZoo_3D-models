/*This software has been adapted from Adafruit for a 3d printed syring pump on Thingaverse.



  This is a test sketch for the Adafruit assembled Motor Shield for Arduino v2
  It won't work with v1.x motor shields! Only for the v2's with built in PWM
  control

  For use with the Adafruit Motor Shield v2
  ---->  http://www.adafruit.com/products/1438
*/
// April 12 2019

#include <Wire.h>
#include <Adafruit_MotorShield.h>

// Create the motor shield object with the default I2C address
Adafruit_MotorShield AFMS = Adafruit_MotorShield();
// Or, create it with a different I2C address (say for stacking)
// Adafruit_MotorShield AFMS = Adafruit_MotorShield(0x61);

// Connect a stepper motor with 200 steps per revolution (1.8 degree)
// to motor port #2 (M3 and M4)
Adafruit_StepperMotor *myMotor = AFMS.getStepper(200, 2);
const int FWDPin = 2;         // the number of the FWD pushbutton pin
const int BACKPin = 3;        // the number of the BACK pushbutton pin
          // One turn of the syringe moter dispenses 200 microliters.
          // This dispensed number is not very accurately found so
          // do not depend on it. I do not have a way to accurately 
          // measure such a small amount.

const int NumberOfSteps = 200;    //This is a number I used for my tests. Change for the amount you will use.
int Flag = 0;
void setup() {
  Serial.begin(9600);             // set up Serial library at 9600 bps
  Serial.println("Stepper test!");

  AFMS.begin();       // create with the default frequency 1.6KHz
  //AFMS.begin(1000);       // OR with a different frequency, say 1KHz

  myMotor->setSpeed(50);      // 50 rpm
  pinMode(FWDPin, INPUT);
  pinMode(BACKPin, INPUT);
}

void loop() {
  int buttonState1 = digitalRead(FWDPin); //read the button pins
  int buttonState2 = digitalRead(BACKPin);
  if ((buttonState1 == 0) && (buttonState2 == 0))
  {
    Serial.println("Button Error!");
  myMotor->release();
    exit(0);            // Exit the program if both buttons are pushed. Wiring error maybe. 
  }
  Serial.println(buttonState1);     // Show the button state. Just for testing, may be commented out.
  Serial.println(buttonState2);
  myMotor->setSpeed(50);        // 50 rpm to home the syringe for refilling
  while (buttonState1 == 0) {     // While forward button is pushed
    myMotor->step(1, FORWARD, DOUBLE);
    buttonState1 = digitalRead(FWDPin);
    Flag = 1;
  myMotor->release();                       // Do not hold the motor still
  }                                         // The gear ratio is hig enough to not require holding.

  while (buttonState2 == 0) {
    myMotor->step(1, BACKWARD, DOUBLE);
    buttonState2 = digitalRead(BACKPin);
    Flag = 1;
  myMotor->release();
  }

   Serial.println("Double coil steps");
  if (Flag != 1)        // if neither button is pushed slow the rate down and take the number of steps
            // called out in NumberOfSteps above.
  {
    myMotor->setSpeed(10);        // 10 rpm - slow the motor down for depositing.
    myMotor->step(NumberOfSteps, FORWARD, DOUBLE); // Forward correct direction.
  }
  if ((buttonState1 == 1) && (buttonState2 == 1))
  {
   myMotor->release();
    exit(0);            // if neither button is pushed and the NumberOfSteps
            // is completed then exit program.
            // Just exit for now. Reset the processor for next deposit.
  }
 
}
